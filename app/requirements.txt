# transformers/tokenizers arrive transitively via mlx-lm, but they are the
# single most drift-prone part of the import chain: `import mlx_lm` imports
# transformers at top level, and an incompatible combo fails with a cryptic
# "'str' object has no attribute '__module__'" at import time. Pinning floors
# to the verified major keeps a stale resolver from pulling an incompatible
# transformers. The exact versions live in requirements.lock.txt.
transformers>=5.12

# vision models route to mlx-vlm; text-only models keep loading via mlx-lm, so
# this is purely additive to the existing chat path.
